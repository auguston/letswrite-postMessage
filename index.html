mother.html

<iframe src="http://iframe.html"></iframe>
<script>
// 這邊要使用的是 iframe onload，用 window onload會造成錯誤，因為iframe有可能未讀取完
$('iframe').on('load', function() {
// 收到message時要執行的function
function receiveMessage(e) {
// 檢查來源網址是否正確
if (e.origin !== "http://iframe.html") {
alert('資料來源錯誤');
return false;
}
else {
// e.data 就是 iframe 中的 'something msg'
var data = e.data;
}
}
// 監聽 message 事件，收到事件時就執行 receiveMessage 這個function
window.addEventListener('message', receiveMessage);
});
</script>
